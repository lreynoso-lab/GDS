ages = [18, 19, 21, 26, 26, 30, 32, 38, 45, 55]

bins = [0, 10, 20, 30, 40, 50, 60]
plt.hist(ages, bins = bins, edgecolor = "black")
plt.title("Edad de los encuestados")
